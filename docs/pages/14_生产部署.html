💻 FastAPI 封装

使用 FastAPI 构建 RESTful API，支持流式输出（SSE）和多会话管理：

# app/main.py
import uuid
from fastapi import FastAPI
from fastapi.responses import StreamingResponse
from fastapi.staticfiles import StaticFiles
from pydantic import BaseModel
from app.rag.chain import RAGChain
from app.models.session import SessionManager

app = FastAPI(title="RAG API", version="1.0")
rag_chain = RAGChain()
session_mgr = SessionManager()

# 挂载静态文件（前端页面）
app.mount("/static", StaticFiles(directory="static"), name="static")

class AskRequest(BaseModel):
    question: str
    session_id: str | None = None  # 可选，不提供则创建新会话

class AskResponse(BaseModel):
    answer: str
    sources: list[str]
    session_id: str

@app.post("/api/ask", response_model=AskResponse)
async def ask(request: AskRequest):
    """非流式问答"""
    # 获取或创建会话
    session_id = request.session_id or str(uuid.uuid4())
    memory = session_mgr.get(session_id)

    # 执行 RAG
    result = rag_chain.ask(
        question=request.question,
        memory=memory
    )

    # 保存会话
    session_mgr.save(session_id, memory)

    return AskResponse(
        answer=result["answer"],
        sources=result["sources"],
        session_id=session_id
    )

@app.post("/api/ask/stream")
async def ask_stream(request: AskRequest):
    """流式问答（SSE）"""
    session_id = request.session_id or str(uuid.uuid4())
    memory = session_mgr.get(session_id)

    async def generate():
        async for chunk in rag_chain.ask_stream(
            question=request.question,
            memory=memory
        ):
            yield f"data: {chunk}\n\n"
        yield "data: [DONE]\n\n"

    return StreamingResponse(
        generate(),
        media_type="text/event-stream",
        headers={"X-Session-Id": session_id}
    )

@app.get("/api/sessions")
async def list_sessions():
    """列出所有会话"""
    return session_mgr.list_all()

@app.delete("/api/sessions/{session_id}")
async def delete_session(session_id: str):
    """删除会话"""
    session_mgr.delete(session_id)
    return {"status": "ok"}

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8000)

🌊 流式输出（SSE）

流式输出让用户体验更好——不需要等待完整回答，看到文字一个一个"蹦出来"。核心技术是 Server-Sent Events（SSE）：

# app/rag/chain.py
from langchain_core.callbacks import AsyncCallbackHandler

class StreamingCallback(AsyncCallbackHandler):
    """流式输出回调处理器"""

    def __init__(self):
        self.chunks = []

    async def on_llm_new_token(self, token: str, **kwargs):
        self.chunks.append(token)

class RAGChain:
    def __init__(self):
        self.retriever = self._build_retriever()
        self.llm = self._build_llm()

    async def ask_stream(self, question: str, memory=None):
        """流式生成回答"""
        # 1. 检索（非流式，必须先完成）
        docs = self.retriever.invoke(question)
        context = "\n".join(d.page_content for d in docs[:5])

        # 2. 构建 Prompt
        prompt = f"""根据以下文档回答问题。

文档：{context}

问题：{question}
回答："""

        # 3. 流式生成
        callback = StreamingCallback()
        async for chunk in self.llm.astream(prompt, callbacks=[callback]):
            yield chunk.content

    def ask(self, question: str, memory=None) -> dict:
        """非流式问答"""
        docs = self.retriever.invoke(question)
        context = "\n".join(d.page_content for d in docs[:5])

        prompt = f"""根据以下文档回答问题。

文档：{context}

问题：{question}
回答："""
        answer = self.llm.invoke(prompt).content

        return {
            "answer": answer,
            "sources": [d.page_content for d in docs[:5]]
        }

💬 多会话管理

生产环境中，多个用户同时使用系统，每个用户的对话历史必须隔离：

# app/models/session.py
import time
from collections import OrderedDict
from langchain_core.messages import HumanMessage, AIMessage

class Session:
    """单个会话"""
    def __init__(self):
        self.history = []       # [(human_msg, ai_msg), ...]
        self.created_at = time.time()
        self.last_active = time.time()

    def add(self, human_msg: str, ai_msg: str):
        self.history.append((human_msg, ai_msg))
        self.last_active = time.time()
        # 只保留最近 10 轮
        if len(self.history) > 10:
            self.history = self.history[-10:]

    def get_context(self) -> str:
        """获取对话上下文"""
        if not self.history:
            return ""
        lines = []
        for h, a in self.history[-3:]:  # 最近 3 轮
            lines.append(f"用户：{h}")
            lines.append(f"AI：{a}")
        return "\n".join(lines)

class SessionManager:
    """会话管理器"""

    def __init__(self, max_sessions=1000, ttl=3600):
        self.sessions: OrderedDict[str, Session] = OrderedDict()
        self.max_sessions = max_sessions
        self.ttl = ttl  # 会话过期时间（秒）

    def get(self, session_id: str) -> Session:
        if session_id not in self.sessions:
            self.sessions[session_id] = Session()
        # LRU：移到最后
        self.sessions.move_to_end(session_id)
        # 清理过期会话
        self._cleanup()
        return self.sessions[session_id]

    def save(self, session_id: str, session: Session):
        self.sessions[session_id] = session

    def delete(self, session_id: str):
        self.sessions.pop(session_id, None)

    def list_all(self) -> list[dict]:
        return [
            {
                "session_id": sid,
                "created_at": s.created_at,
                "last_active": s.last_active,
                "turns": len(s.history),
            }
            for sid, s in self.sessions.items()
        ]

    def _cleanup(self):
        """清理过期和超出限制的会话"""
        now = time.time()
        # 清理过期
        expired = [
            sid for sid, s in self.sessions.items()
            if now - s.last_active > self.ttl
        ]
        for sid in expired:
            del self.sessions[sid]
        # 清理超出数量限制
        while len(self.sessions) > self.max_sessions:
            self.sessions.popitem(last=False)
生产环境建议：用 Redis 替代内存存储会话，支持分布式部署和持久化
会话 TTL：建议设置 1 小时过期，避免长期不活跃的会话占用内存

        const sessionId = localStorage.getItem('session_id') || '';
        const API_BASE = '';

        function addMessage(role, content, sources=[]) {
            const div = document.createElement('div');
            div.className = 'message ' + role;
            const avatar = role === 'assistant' ? 'AI' : '我';
            let html = `<div class="avatar">${avatar}</div>`;
            html += `<div class="bubble">${content}`;
            if (sources.length) {
                html += `<div class="sources">📄 来源：${sources.join(' | ')}</div>`;
            }
            html += '</div>';
            div.innerHTML = html;
            document.getElementById('messages').appendChild(div);
        }

        async function ask() {
            const input = document.getElementById('question');
            const question = input.value.trim();
            if (!question) return;

            addMessage('user', question);
            input.value = '';
            document.getElementById('sendBtn').disabled = true;

            // 添加 AI 占位
            const aiDiv = document.createElement('div');
            aiDiv.className = 'message assistant';
            aiDiv.innerHTML = '<div class="avatar">AI</div><div class="bubble" id="ai-typing">思考中...</div>';
            document.getElementById('messages').appendChild(aiDiv);

            try {
                const resp = await fetch(`${API_BASE}/api/ask/stream`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ question, session_id: sessionId || undefined })
                });

                const reader = resp.body.getReader();
                const decoder = new TextDecoder();
                let answer = '';

                while (true) {
                    const { done, value } = await reader.read();
                    if (done) break;
                    const text = decoder.decode(value);
                    for (const line of text.split('\n\n')) {
                        if (line.startsWith('data: ') && line !== 'data: [DONE]') {
                            const chunk = line.slice(6);
                            answer += chunk;
                            document.getElementById('ai-typing').textContent = answer;
                        }
                    }
                }

                // 保存 session_id
                const sid = resp.headers.get('X-Session-Id');
                if (sid) {
                    localStorage.setItem('session_id', sid);
                }
            } catch (e) {
                document.getElementById('ai-typing').textContent = '请求失败：' + e.message;
            }

            document.getElementById('sendBtn').disabled = false;
        }
    </script>

import requests
import json

BASE_URL = "http://localhost:8000"

def test_non_streaming():
    """测试非流式问答"""
    resp = requests.post(f"{BASE_URL}/api/ask", json={
        "question": "谢初的官职是什么？"
    })
    data = resp.json()
    print(f"回答：{data['answer']}")
    print(f"来源：{data['sources']}")
    print(f"会话 ID：{data['session_id']}")
    return data["session_id"]

def test_streaming():
    """测试流式问答"""
    resp = requests.post(f"{BASE_URL}/api/ask/stream", json={
        "question": "沈令月是谁？"
    }, stream=True)

    print("流式输出：", end="", flush=True)
    for line in resp.iter_lines():
        if line:
            text = line.decode("utf-8")
            if text.startswith("data: ") and text != "data: [DONE]":
                chunk = text[6:]
                print(chunk, end="", flush=True)
    print()

def test_multi_turn():
    """测试多轮对话"""
    # 第一轮
    resp = requests.post(f"{BASE_URL}/api/ask", json={

🚀 快速开始

1. 安装依赖
pip install fastapi uvicorn langchain chromadb
2. 准备文档
将文档放入 data/documents/
3. 构建向量库
python -m app.rag.build_index
4. 启动服务
uvicorn app.main:app --host 0.0.0.0 --port 8000
5. 访问
浏览器打开 http://localhost:8000/static/index.html
